/**
 * Landing Component
 * @param {string} title
 * @param {string} body
 * @param {string} avatarUrl
 * @param {string} cover
 */
export default class Landing extends Component {
  state = {
    news: [],
  };

  componentDidMount() {
    const { news } = this.props;
    this.updateNews(news);
  }

  updateNews = (news) => {
    this.setState({ news });
  };

  render() {
    const { news } = this.state;
    const newsCards = news.map((item) => (
      <LandingCard
        key={item.key}
        title={item.title}
        body={item.body}
        avatarUrl={item.avatarUrl}
        cover={item.cover}
      />
    ));
    return newsCards;
  }
}

Landing.propTypes = {
  news: PropType.array, // isRequired
};

Landing.defaultProps = {
  news: newsItems,
};
